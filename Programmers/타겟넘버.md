    answer = 0
    n = len(numbers)
    
    def dfs(res, i):
        if i==n:
            if res == target:
                nonlocal answer
                answer += 1
            return
        else:
            dfs(res+numbers[i], i+1)
            dfs(res-numbers[i], i+1)
    
    dfs(0,0)
    
    return answer
