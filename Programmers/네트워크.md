```
def solution(n, computers):
    answer = 0
    visited = [0] * n
    
    def dfs(v):
        visited[v] = 1
        for i in range(n):
            if not visited[i] and computers[v][i] == 1:
                dfs(i)
                
    
    for i in range(n):
        if not visited[i]:
            dfs(i)
            answer += 1
            
    return answer
```
