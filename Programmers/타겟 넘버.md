1. DFS
```
def solution(numbers, target):
    answer = 0
    n = len(numbers)
    
    def dfs(res, i):
        if i==n:
            if res == target:
                nonlocal answer
                answer += 1
            return
        else:
            dfs(res+numbers[i], i+1)
            dfs(res-numbers[i], i+1)
    
    dfs(0,0)
    
    return answer
```

2. BFS
```
from collections import deque

def solution(numbers, target):
    answer = 0
    n = len(numbers)
    q = deque()
    q.append([numbers[0], 0])
    q.append([-1*numbers[0],0])

    while q:
        temp, i = q.popleft()
        i += 1
        if i<n:
            q.append([temp+numbers[i], i])
            q.append([temp-numbers[i], i])
        else:
            if temp == target:
                answer += 1
    return answer
```
